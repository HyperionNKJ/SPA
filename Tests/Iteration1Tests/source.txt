procedure Example {
  x = 2;					// 1
  z = 3;					// 2
  i = 5;					// 3
  while (i!=0) {			// 4
    x = x - 1;				// 5
    if (x==1) then {		// 6
      z = x + 1; }			// 7
    else {
      y = z + x; }			// 8
    z = z + x + i;			// 9
    call q;					// 10
    i = i - 1; }			// 11
  call p; }					// 12

procedure p {
  if (x<0) then {			// 13
    while (i>0) {			// 14
      x = z * 3 + 2 * y;	// 15	
      call q;				// 16
      i = i - 1; }			// 17
    x = x + 1;				// 18
    z = x + z; }			// 19
  else {
    z = 1; }				// 20
  z = z + x + i; }			// 21

procedure q {
  if (x==1) then {			// 22
    z = x + 1; }			// 23
  else {
    x = z + x; } }			// 24
